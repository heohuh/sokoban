int get_box_coordinate( int level ) //상자의 위치를 찾는 함수. - 윤동우
{
	int i = 0;
	int j = 0;
	box_count = 0;

	for( i = 0; i < MAXLINE; i++)          //전체 배열을 다 돌면서 박스 정보를 저장한다 - 윤동우
	{
		for( j = 0; j < MAXLINE; j++)
		{
			if( map[level][i][j] == 'O' )
			{
				box_coord[box_count][0] = i;
				box_coord[box_count][1] = j;
				box_count++;
			}
		}
	}
	return box_count;
}
