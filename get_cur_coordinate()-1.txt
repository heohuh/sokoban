void get_cur_coordinate()     //플레이어('@')의 위치를 찾는 함수 - 윤동우
{
	int i = 0;
	int j = 0;

	for( i = 0; i < MAXLINE; i++ )
	{
		for( j = 0; j < MAXWIDTH; j++ )
		{
			if( cur_map[i][j] == '@' )
			{
				cur_coord[0] = i; cur_coord[1] = j;
				break;
			}                          //전체 배열을 다 돌면서 '@'문자 정보의 위치를 저장. - 윤동우
		}
	}
}
