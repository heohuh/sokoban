int get_gold_coordinate()     //금의 위치를 찾는 함수 - 윤동우
{
	int i = 0;
	int j = 0;
	int gold_count = 0;

	for( i = 0; i < MAXLINE; i++)           //전체 배열을 다 돌면서 금의 정보를 저장한다 - 윤동우
	{
		for( j = 0; j < MAXLINE; j++)
		{
			if( cur_map[i][j] == '$' )
			{
				gold_coord[gold_count][0] = i;
				gold_coord[gold_count][1] = j;
				gold_count++;
			}
		}
	}
	return gold_count;
}
